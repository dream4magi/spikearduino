[BASIC INFORMATION]
PORT RANGE 1~50
MAPPING PIN = PIN2~PIN51
ON = 1
OFF = 0
BAUD:115200


[input format]

A[PORT_NO][ON/OFF]Z
EX:
 A11Z = PORT 1 ON
 A10Z = PORT 1 OFF
 A501Z = PORT 50 ON
 A500Z = PORT 50 OFF

SPECIAL:
 A00Z = ALL PORT OFF
 A01Z = ALL PORT ON




IF PIN52 = HIGH
Turn all on
